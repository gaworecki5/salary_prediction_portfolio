def plot_feature(df, col):
    '''
    Make plot for each features
    left, the distribution of samples on the feature
    right, the dependance of salary on the feature
    '''
    plt.figure(figsize = (14, 6))
    plt.subplot(1, 2, 1)
    if df[col].dtype == 'int64':
        df[col].value_counts().sort_index().plot()
    else:
        #change the categorical variable to category type and order their level by the mean salary
        #in each category
        mean = df.groupby(col)['salary'].mean()
        #df[col] = df[col].astype('category')
        levels = mean.sort_values().index.tolist()
        df[col].cat.reorder_categories(levels, inplace=True)
        df[col].value_counts().plot()
    plt.xticks(rotation=45)
    plt.xlabel(col)
    plt.ylabel('Counts')
    plt.subplot(1, 2, 2)

    if df[col].dtype == 'int64' or col == 'companyId':
        #plot the mean salary for each category and fill between the (mean - std, mean + std)
        mean = df.groupby(col)['salary'].mean()
        std = df.groupby(col)['salary'].std()
        mean.plot()
        plt.fill_between(range(len(std.index)), mean.values-std.values, mean.values + std.values, \
                         alpha = 0.1)
    else:
        sns.boxplot(x = col, y = 'salary', data=df)
    
    plt.xticks(rotation=45)
    plt.ylabel('Salaries')
    plt.show()

#baseline model

def naive_encode(df, col, new_col ='base_predict'):
    ''' create new column in df for baseline predictions'''
    cat_dict ={}
    cat_salaries = df.groupby(col).salary.mean()
    cats = df[col].cat.categories.tolist()
    for cat in cats:
        cat_dict[cat] = cat_salaries[cat]  
    df[new_col] = df[col].map(cat_dict)

def run_model(model ,cv=5, scorer = 'neg_mean_squared_error', n_jobs =2):
    ''' function to run model and output cross-validated error metric stats and run time'''
    start = time.time()
    model_score = model_select.cross_val_score(model, features, target, cv=cv, scoring = scorer, n_jobs =n_jobs)
    run_time = time.time() - start
    print('{} model - average MSE: {:06.2f}, Stdev: {:05.2f}  (5_fold cross-validation)'.
          format(str(model).split('(')[0], -(model_score.mean()), model_score.std()))
    print('Model CV run time : {:06.2f} seconds'.format(run_time))

def load_data(data_file):
    '''function to read a csv into a df'''
    return pd.read_csv(data_file)

def merge_clean_data(features_df, target_df):
    '''function to merge features df with target df'''
    #merge features and target
    train_df = features_df.merge(target_df, on = 'jobId')
    #remove outliers with incorrect 0 salary
    train_df = train_df.loc[train_df.salary > 8.5]
    #change column type to categorical
    for col in list(train_df.select_dtypes('object').columns):
        train_df[col] = train_df[col].astype('category')
    return train_df
    
def feature_encode_scale(train_df):
    '''function to encode features:
    1. encode degree ordinaly
    2. drop low correlation features jobId and companyId
    3. encode other categoricals as one-hot
    4. scale numeric cols'''
    
    # ecode degree column ordinal
    degree_ord_map = {'NONE': 0.0, 'HIGH_SCHOOL': 1, 'BACHELORS': 2, 
                      'MASTERS': 3, 'DOCTORAL': 4}
    train_df['degree'] = train_df['degree'].map(degree_ord_map)
    
    #drop columns with low correlation
    train_df = train_df.drop(columns=['jobId','companyId'])
    
    # encode remaining categoricals as one-hot
    one_hot_columns = ['jobType','major','industry']
    train_df = pd.get_dummies(train_df, columns=one_hot_columns)
    
    #scale numeric cols with StandardScalerScaler
    number_cols = ['milesFromMetropolis','yearsExperience']
    for col in number_cols:
        scaler = preproc.StandardScaler()
        train_df[col] = scaler.fit_transform(train_df[[col]])
    return train_df
       
def train_model(model, train_df):
    '''function to train model on df with features and target variables'''
    #split train_df into target_df and features_df
    target_df = train_df.loc[:, 'salary']
    features_df = train_df.drop(columns=['salary'])
    model.fit(features_df, target_df) 

def get_predictions(model, test_df):
    '''function to preprocess test_df and make predictions from features'''
    predict_df = feature_encode_scale(test_df)
    predictions = model.predict(predict_df)
    return predictions
